🔧 QUICK CONFIGURATION:

  # Auto Everything (Recommended)
  MODEL_NAME=microsoft/Phi-3-mini-4k-instruct
  MODEL_TYPE=auto
  QUANTIZATION=auto
  
  # CPU-Optimized (Fastest)
  MODEL_NAME=TheBloke/Phi-3-mini-4k-instruct-GGUF
  MODEL_TYPE=gguf
  QUANTIZATION=4bit
  
  # GPU Production (Best Performance)
  MODEL_NAME=microsoft/Phi-3-mini-4k-instruct
  MODEL_TYPE=vllm
  USE_GPU=True
  
  # Large Model (7B)
  MODEL_NAME=mistralai/Mistral-7B-Instruct-v0.2
  MODEL_TYPE=airllm
  QUANTIZATION=4bit

💻 USAGE EXAMPLE:

  from app.services.llm_service import llm_service
  
  # Load model (auto-optimized)
  await llm_service.load_model()
  
  # Generate response
  response = await llm_service.generate("Explain AI")
  
  # Stream tokens
  async for token in llm_service.generate_stream("Story"):
      print(token, end='', flush=True)
  
  # Check config
  info = llm_service.get_model_info()
  print(info)  # Shows backend, quantization, etc.
